        <StackPanel>
            <Ellipse x:Name="Light" Width="200" Height="200" HorizontalAlignment="Center"
                     Fill="Red" />
            <Button Content="Change Lights" Width="200" Height="200" HorizontalAlignment="Center" Click="Button_Click" />
        </StackPanel>
        
        
        private void Button_Click(object sender, RoutedEventArgs e)
        {
            var fill = Light.Fill as SolidColorBrush;

            if (fill.Color == Colors.Red)
            {
                Light.Fill = new SolidColorBrush(Colors.Green);
            }
            else
            {
                Light.Fill = new SolidColorBrush(Colors.Red);
            }
        }
